cmake_minimum_required(VERSION 3.30)
project(El_Creatura C)

set(CMAKE_C_STANDARD 23)
set(CMAKE_C_STANDARD_REQUIRED ON)
if(STATIC_PROJECT)
    set(SDL_STATIC ON)
    set(SDL_TEST OFF)
    set(BUILD_SHARED_LIBS OFF)
    add_subdirectory(vendor)
endif ()
add_executable(El_Creatura main.c)
if(STATIC_PROJECT)
    target_link_libraries(El_Creatura PRIVATE vendor)
else ()
    target_link_libraries(El_Creatura PRIVATE SDL3)
    target_link_libraries(El_Creatura PRIVATE SDL3_image)
    target_link_libraries(El_Creatura PRIVATE SDL3_ttf)
endif ()
